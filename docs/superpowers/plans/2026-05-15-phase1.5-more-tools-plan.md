```ts
/**
 * A text-span edit against a statement's raw payload. The Phase 1 identifier
 * mutation is the degenerate case: start = offset, end = offset + oldText.length.
 * A pure insertion is a zero-width span (start === end, oldText === "").
 */
export interface TextSpanEdit {
  start: number;
  end: number;
  oldText: string;
  newText: string;
}

/**
 * Back-compat alias. Phase 1 callers constructed { offset, oldText, newText };
 * keep that shape expressible by mapping offset -> a span. New code uses
 * TextSpanEdit directly.
 */
export interface IdentifierMutation {
  offset: number;
  oldText: string;
  newText: string;
}

export function identifierMutationToSpan(m: IdentifierMutation): TextSpanEdit {
  return {
    start: m.offset,
    end: m.offset + m.oldText.length,
    oldText: m.oldText,
    newText: m.newText
  };
}

export function spliceStatement(
  payload: string,
  edits: TextSpanEdit[]
): string {
  if (edits.length === 0) {
    return payload;
  }

  // Descending by start so earlier edits' offsets stay valid as we mutate
  // from the back. Same algorithm Phase 1 used for identifier mutations.
  const sorted = [...edits].sort((left, right) => right.start - left.start);
  let out = payload;

  for (const edit of sorted) {
    const actual = out.slice(edit.start, edit.start + edit.oldText.length);
    if (actual !== edit.oldText || edit.end !== edit.start + edit.oldText.length) {
      throw new Error(
        `oldText mismatch at [${edit.start},${edit.end}): expected ${JSON.stringify(
          edit.oldText
        )}, got ${JSON.stringify(actual)}`
      );
    }
    out = out.slice(0, edit.start) + edit.newText + out.slice(edit.end);
  }

  return out;
}
```

Modify `/Users/toddhebebrand/Strata/packages/render/src/index.ts`. Change the import line and `RenderOverlay`, and the `mutationsForNode` assembly so identifier mutations are converted to spans and concatenated with any per-statement `textSpanMutations`:
```ts
import type { NodeRow } from "@strata/store";
import {
  spliceStatement,
  identifierMutationToSpan,
  type TextSpanEdit
} from "./splice";

export interface SourceMapEntry {
  renderedStart: number;
  renderedEnd: number;
  nodeId: string;
}

export interface RenderOverlay {
  identifierMutations: Map<string, { text: string }>;
  /** statementId -> text-span edits against that statement's raw payload. */
  textSpanMutations?: Map<string, TextSpanEdit[]>;
}

export interface RenderResult {
  text: string;
  sourceMap: SourceMapEntry[];
}

export function render(module: NodeRow, children: NodeRow[]): string {
  return renderWithSourceMap(module, children).text;
}

export function renderWithSourceMap(
  _module: NodeRow,
  children: NodeRow[],
  overlay?: RenderOverlay
): RenderResult {
  const topLevel = [...children]
    .filter((child) => child.kind !== "Identifier" && child.childIndex !== null)
    .sort((left, right) => (left.childIndex ?? 0) - (right.childIndex ?? 0));

  const identifiersByParent = new Map<string, NodeRow[]>();
  for (const child of children) {
    if (child.kind !== "Identifier" || !child.parentId) {
      continue;
    }
    const siblings = identifiersByParent.get(child.parentId) ?? [];
    siblings.push(child);
    identifiersByParent.set(child.parentId, siblings);
  }

  const sourceMap: SourceMapEntry[] = [];
  const parts: string[] = [];
  let cursor = 0;

  for (const node of topLevel) {
    const edits = editsForNode(
      node.id,
      identifiersByParent.get(node.id) ?? [],
      overlay
    );
    const text = spliceStatement(node.payload, edits);
    const renderedStart = cursor;
    cursor += text.length;
    parts.push(text);
    sourceMap.push({ renderedStart, renderedEnd: cursor, nodeId: node.id });
  }

  return { text: parts.join(""), sourceMap };
}

function editsForNode(
  statementId: string,
  identifiers: NodeRow[],
  overlay: RenderOverlay | undefined
): TextSpanEdit[] {
  if (!overlay) {
    return [];
  }

  const edits: TextSpanEdit[] = [];
  for (const identifier of identifiers) {
    const updated = overlay.identifierMutations.get(identifier.id);
    if (!updated) {
      continue;
    }
    const payload = JSON.parse(identifier.payload) as {
      text: string;
      offset: number;
    };
    edits.push(
      identifierMutationToSpan({
        offset: payload.offset,
        oldText: payload.text,
        newText: updated.text
      })
    );
  }

  for (const edit of overlay.textSpanMutations?.get(statementId) ?? []) {
    edits.push(edit);
  }

  return edits;
}
```

```ts
import ts from "typescript";

export type SpanKind = "params" | "returnType" | "body";

export interface Span {
  /** Offset into the statement's raw payload (inclusive). */
  start: number;
  /** Offset into the statement's raw payload (exclusive). */
  end: number;
  /** Convenience: payload.slice(start, end). Empty for an insertion span. */
  text: string;
}

/**
 * Re-parse a single statement's raw payload (which IS verbatim source —
 * statement.getFullText) and return the payload-relative span of its
 * parameter list, return-type annotation, or body. Public TypeScript APIs
 * only: ts.createSourceFile + getChildren walk. No ingest lowering, no
 * internal TS properties (decisions.md 2026-05-15 BS1 discipline).
 */
export function locateSpan(payload: string, kind: SpanKind): Span {
  const sf = ts.createSourceFile(
    "__span__.ts",
    payload,
    ts.ScriptTarget.Latest,
    /* setParentNodes */ true,
    ts.ScriptKind.TS
  );

  const fn = sf.statements.find((s): s is ts.FunctionDeclaration =>
    ts.isFunctionDeclaration(s)
  );
  if (!fn) {
    throw new Error(
      "spanReparse: payload is not a function declaration statement"
    );
  }

  if (kind === "params") {
    // The NodeArray<ParameterDeclaration> has .pos/.end; .pos points just
    // after the open paren (includes leading trivia), so use the first
    // parameter's getStart() when present, else the position just inside `(`.
    const params = fn.parameters;
    if (params.length > 0) {
      const start = params[0]!.getStart(sf);
      const end = params[params.length - 1]!.getEnd();
      return { start, end, text: payload.slice(start, end) };
    }
    // Empty param list: insertion point just inside the open paren.
    const openParen = findToken(fn, sf, ts.SyntaxKind.OpenParenToken);
    const at = openParen.getEnd();
    return { start: at, end: at, text: "" };
  }

  if (kind === "returnType") {
    if (fn.type) {
      const start = fn.type.getStart(sf);
      const end = fn.type.getEnd();
      return { start, end, text: payload.slice(start, end) };
    }
    // No annotation: insertion point immediately after the param list `)`.
    const closeParen = findToken(fn, sf, ts.SyntaxKind.CloseParenToken);
    const at = closeParen.getEnd();
    return { start: at, end: at, text: "" };
  }

  // body
  if (!fn.body) {
    throw new Error("spanReparse: function declaration has no body");
  }
  const start = fn.body.getStart(sf);
  const end = fn.body.getEnd();
  return { start, end, text: payload.slice(start, end) };
}

function findToken(
  parent: ts.Node,
  sf: ts.SourceFile,
  kind: ts.SyntaxKind
): ts.Node {
  for (const child of parent.getChildren(sf)) {
    if (child.kind === kind) {
      return child;
    }
  }
  throw new Error(
    `spanReparse: token ${ts.SyntaxKind[kind]} not found on declaration`
  );
}
```

```ts
import type { Db } from "./schema";
import { locateSpan } from "./spanReparse";
import { queuePendingOp, queueTextSpanEdit, type TxHandle } from "./transactions";

function isValidType(typeText: string): boolean {
  const sf = ts.createSourceFile(
    "__t__.ts",
    `let __x__: ${typeText};`,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TS
  );
  const stmt = sf.statements[0];
  if (!stmt || !ts.isVariableStatement(stmt)) return false;
  const decl = stmt.declarationList.declarations[0];
  // No parser diagnostics on the source file means it parsed; require a type.
  return Boolean(decl && decl.type) && !hasParseError(sf);
}

function hasParseError(sf: ts.SourceFile): boolean {
  // Public surface: parseDiagnostics is exposed on SourceFile as
  // (sf as { parseDiagnostics?: ts.Diagnostic[] }). Avoid it — instead
  // re-stringify check: a well-formed type round-trips. Use a printer.
  const printer = ts.createPrinter();
  try {
    printer.printFile(sf);
    return false;
  } catch {
    return true;
  }
}
```

```ts
import ts from "typescript";
import { findNodeById } from "./nodes";
import type { Db } from "./schema";
import { locateSpan } from "./spanReparse";
import { queuePendingOp, queueTextSpanEdit, type TxHandle } from "./transactions";

/**
 * Syntactic-only check: wrap as a function and confirm the new body parses
 * as a single ts.Block. Semantic correctness is the compiler's job at
 * validate (the consistent Phase 1 stance), not pre-checked here.
 */
function isValidBlock(bodyText: string): boolean {
  const sf = ts.createSourceFile(
    "__b__.ts",
    `function __probe__() ${bodyText}`,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TS
  );
  const fn = sf.statements[0];
  if (!fn || !ts.isFunctionDeclaration(fn) || !fn.body) {
    return false;
  }
  if (!ts.isBlock(fn.body)) {
    return false;
  }
  // The body must consume the whole provided text (no trailing garbage):
  // its end must reach the end of the wrapped source minus nothing extra.
  const wrappedPrefix = "function __probe__() ".length;
  return fn.body.getEnd() === wrappedPrefix + bodyText.length;
}

export function replace_body(
  db: Db,
  tx: TxHandle,
  functionId: string,
  newBody: string
): void {
  const declaration = findNodeById(db, functionId);
  if (!declaration) {
    throw new Error(`Declaration not found: ${functionId}`);
  }
  if (declaration.kind !== "FunctionDeclaration") {
    throw new Error(
      `Node ${functionId} is not a FunctionDeclaration (kind=${declaration.kind})`
    );
  }
  if (!isValidBlock(newBody)) {
    throw new Error(
      `Invalid body: must be a syntactically valid { ... } block: ` +
        JSON.stringify(newBody.slice(0, 40))
    );
  }

  const span = locateSpan(declaration.payload, "body");
  if (span.text === newBody) {
    return; // no-op
  }

  queueTextSpanEdit(tx, functionId, {
    start: span.start,
    end: span.end,
    oldText: span.text,
    newText: newBody
  });

  queuePendingOp(tx, {
    kind: "ReplaceBody",
    paramsJson: JSON.stringify({
      function_id: functionId,
      new_body_len: newBody.length
    }),
    affectedNodeIdsJson: JSON.stringify([functionId]),
    reasoning: null
  });
}

export const replaceBody = replace_body;
```

```ts
import ts from "typescript";
import { findNodeById, listChildren } from "./nodes";
import { getReferencesByTo } from "./references";
import type { Db } from "./schema";

export interface Callsite {
  /** Statement node ID whose payload contains the call. */
  statementId: string;
  /** Payload offset just inside the call's argument list (after `(` or
   * after the last existing argument), where a new argument is inserted. */
  argListInsertOffset: number;
  /** Number of existing arguments at this callsite. */
  existingArgCount: number;
}

export interface NonCallReference {
  statementId: string;
  /** Why it is not an argument-insertion site (HOF value, alias, type pos). */
  shape: "value-position" | "no-enclosing-call";
}

export interface CallsiteResolution {
  callsites: Callsite[];
  /** Reference identifiers with no enclosing call-whose-callee-is-them.
   * These become wrong-arity sites the compiler flags — NOT mis-edited. */
  nonCallReferences: NonCallReference[];
}

/**
 * BS15-B surface. The node_references index gives reference *identifiers*.
 * For add_parameter a callsite is a ts.CallExpression whose callee resolves
 * to the function. We locate it by re-parsing the referring statement's own
 * stored payload (public ts.createSourceFile + getChildren — BS1 discipline)
 * and walking up from the reference identifier's payload span to the nearest
 * enclosing CallExpression whose expression IS that identifier. Higher-order
 * (.map(fn)) and aliased uses have a reference identifier but no such call:
 * correctly classified as non-argument sites, surfaced for the compiler to
 * flag, never silently mis-edited.
 */
export function resolveCallsites(
  db: Db,
  functionId: string
): CallsiteResolution {
  const declaration = findNodeById(db, functionId);
  if (!declaration) {
    throw new Error(`Declaration not found: ${functionId}`);
  }
  const declIdentifier = listChildren(db, functionId).find(
    (c) => c.kind === "Identifier"
  );
  if (!declIdentifier) {
    throw new Error(`Declaration ${functionId} has no identifier child`);
  }

  const refs = getReferencesByTo(db, declIdentifier.id);
  const callsites: Callsite[] = [];
  const nonCallReferences: NonCallReference[] = [];

  for (const ref of refs) {
    const refNode = findNodeById(db, ref.fromNodeId);
    if (!refNode || !refNode.parentId) {
      continue;
    }
    const statement = findNodeById(db, refNode.parentId);
    if (!statement) {
      continue;
    }
    const refPayload = JSON.parse(refNode.payload) as {
      text: string;
      offset: number;
    };

    const sf = ts.createSourceFile(
      "__cs__.ts",
      statement.payload,
      ts.ScriptTarget.Latest,
      true,
      ts.ScriptKind.TS
    );

    const idNode = findIdentifierAtOffset(
      sf,
      refPayload.offset,
      refPayload.text
    );
    if (!idNode) {
      nonCallReferences.push({
        statementId: statement.id,
        shape: "no-enclosing-call"
      });
      continue;
    }

    const call = enclosingCallWhoseCalleeIs(idNode);
    if (!call) {
      nonCallReferences.push({
        statementId: statement.id,
        shape: "value-position"
      });
      continue;
    }

    const args = call.arguments;
    const insertOffset =
      args.length > 0
        ? args[args.length - 1]!.getEnd()
        : findToken(call, sf, ts.SyntaxKind.OpenParenToken).getEnd();
    callsites.push({
      statementId: statement.id,
      argListInsertOffset: insertOffset,
      existingArgCount: args.length
    });
  }

  return { callsites, nonCallReferences };
}

function findIdentifierAtOffset(
  sf: ts.SourceFile,
  offset: number,
  text: string
): ts.Identifier | undefined {
  let found: ts.Identifier | undefined;
  function visit(node: ts.Node): void {
    if (found) return;
    if (
      ts.isIdentifier(node) &&
      node.text === text &&
      node.getStart(sf) === offset
    ) {
      found = node;
      return;
    }
    for (const child of node.getChildren(sf)) {
      visit(child);
    }
  }
  visit(sf);
  return found;
}

function enclosingCallWhoseCalleeIs(
  id: ts.Identifier
): ts.CallExpression | undefined {
  let node: ts.Node | undefined = id.parent;
  let callee: ts.Node = id;
  // Allow the callee to be the identifier directly, or a parenthesized
  // identifier; do NOT allow property-access/element-access (that is a
  // different symbol) or being an argument (value position).
  while (node) {
    if (ts.isParenthesizedExpression(node) && node.expression === callee) {
      callee = node;
      node = node.parent;
      continue;
    }
    if (ts.isCallExpression(node)) {
      return node.expression === callee ? node : undefined;
    }
    return undefined;
  }
  return undefined;
}

function findToken(
  parent: ts.Node,
  sf: ts.SourceFile,
  kind: ts.SyntaxKind
): ts.Node {
  for (const child of parent.getChildren(sf)) {
    if (child.kind === kind) return child;
  }
  throw new Error(`token ${ts.SyntaxKind[kind]} not found`);
}
```

```ts
import ts from "typescript";
import { resolveCallsites } from "./callsites";
import { findNodeById } from "./nodes";
import type { Db } from "./schema";
import { locateSpan } from "./spanReparse";
import { queuePendingOp, queueTextSpanEdit, type TxHandle } from "./transactions";

const IDENT_PATTERN = /^[A-Za-z_$][A-Za-z0-9_$]*$/;

function parsesAsType(typeText: string): boolean {
  const sf = ts.createSourceFile(
    "__t__.ts",
    `let __x__: ${typeText};`,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TS
  );
  const stmt = sf.statements[0];
  return (
    !!stmt &&
    ts.isVariableStatement(stmt) &&
    !!stmt.declarationList.declarations[0]?.type
  );
}

function parsesAsExpression(exprText: string): boolean {
  const sf = ts.createSourceFile(
    "__e__.ts",
    `const __x__ = (${exprText});`,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TS
  );
  const stmt = sf.statements[0];
  return (
    !!stmt &&
    ts.isVariableStatement(stmt) &&
    !!stmt.declarationList.declarations[0]?.initializer
  );
}

export function add_parameter(
  db: Db,
  tx: TxHandle,
  functionId: string,
  name: string,
  type: string,
  position: number,
  defaultValue?: string
): void {
  if (!IDENT_PATTERN.test(name)) {
    throw new Error(`Invalid TypeScript identifier: ${JSON.stringify(name)}`);
  }
  if (!parsesAsType(type)) {
    throw new Error(`Invalid parameter type: ${JSON.stringify(type)}`);
  }
  if (defaultValue !== undefined && !parsesAsExpression(defaultValue)) {
    throw new Error(`Invalid default expression: ${JSON.stringify(defaultValue)}`);
  }

  const declaration = findNodeById(db, functionId);
  if (!declaration) {
    throw new Error(`Declaration not found: ${functionId}`);
  }
  if (declaration.kind !== "FunctionDeclaration") {
    throw new Error(
      `Node ${functionId} is not a FunctionDeclaration (kind=${declaration.kind})`
    );
  }

  // --- declaration edit ---
  const sf = ts.createSourceFile(
    "__d__.ts",
    declaration.payload,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TS
  );
  const fn = sf.statements.find(ts.isFunctionDeclaration);
  if (!fn) {
    throw new Error("add_parameter: payload is not a function declaration");
  }
  const params = fn.parameters;
  const clamped = Math.max(0, Math.min(position, params.length));
  const paramText =
    defaultValue !== undefined
      ? `${name}: ${type} = ${defaultValue}`
      : `${name}: ${type}`;

  let declStart: number;
  let declNewText: string;
  if (params.length === 0) {
    const span = locateSpan(declaration.payload, "params");
    declStart = span.start; // just inside `(`
    declNewText = paramText;
  } else if (clamped === params.length) {
    declStart = params[params.length - 1]!.getEnd();
    declNewText = `, ${paramText}`;
  } else if (clamped === 0) {
    declStart = params[0]!.getStart(sf);
    declNewText = `${paramText}, `;
  } else {
    declStart = params[clamped - 1]!.getEnd();
    declNewText = `, ${paramText}`;
  }
  queueTextSpanEdit(tx, functionId, {
    start: declStart,
    end: declStart,
    oldText: "",
    newText: declNewText
  });

  // --- callsite edits ---
  const { callsites } = resolveCallsites(db, functionId);
  const affected: string[] = [functionId];
  // The argument value is the caller's decision; the tool reserves the
  // slot with the parameter's default if it has one, else `undefined`,
  // which the agent overwrites per-callsite in the T01 task.
  const slotValue = defaultValue !== undefined ? defaultValue : "undefined";
  for (const cs of callsites) {
    const insertText =
      cs.existingArgCount > 0 ? `, ${slotValue}` : `${slotValue}`;
    queueTextSpanEdit(tx, cs.statementId, {
      start: cs.argListInsertOffset,
      end: cs.argListInsertOffset,
      oldText: "",
      newText: insertText
    });
    if (!affected.includes(cs.statementId)) {
      affected.push(cs.statementId);
    }
  }

  queuePendingOp(tx, {
    kind: "AddParameter",
    paramsJson: JSON.stringify({
      function_id: functionId,
      name,
      type,
      position: clamped,
      has_default: defaultValue !== undefined
    }),
    affectedNodeIdsJson: JSON.stringify(affected),
    reasoning: null
  });
}

export const addParameter = add_parameter;
```
> Position-handling note: callsite slot insertion uses each callsite's own argument-list end / open-paren offset from `resolveCallsites`; the per-callsite *position* matching the declaration `position` is left to the compiler for non-trailing inserts (Phase 1 "trust the compiler" stance, spec § add_parameter edge cases). If the test's `position=1` (trailing) case fails because the param-list insertion offset/trivia is off, adjust against `spanReparse` and re-run; the assertion strings pin the exact expected declaration text.

```ts
 * The text-derived T01 criteria — a PURE function of final module text,
 * keyed by POSIX path relative to corpus src/. Both configs feed this:
 * substrate renders committed modules; baseline reads post-edit .ts.
 * Whitespace-tolerant so render canonicalization vs hand-edit do not
 * diverge (BS15-C). Mirrors evaluateT03TextCriteria exactly in shape.

export function evaluateT01TextCriteria(
  modules: Map<string, string>
): T01TextCriteria {
  const format = mustGet(modules, "lib/format.ts");
  const server = mustGet(modules, "server/events.ts");
  const ui = mustGet(modules, "ui/timeline.ts");

  return {
    signatureHasTimezone:
      /function\s+formatTimestamp\s*\(\s*ts\s*:\s*number\s*,\s*timezone\s*:\s*string/.test(

    defaultIsUtcString: /timezone\s*:\s*string\s*=\s*"UTC"/.test(format),
    // every formatTimestamp(...) call in server/ passes "UTC" as 2nd arg
    serverCallsitesUtc:
      /formatTimestamp\(\s*[^,()]+,\s*"UTC"\s*\)/.test(server) &&
      !/formatTimestamp\(\s*[^,()]+\)\s*[`)]/.test(server),
    // ui direct calls pass "local" (or default); the .map HOF use is left
    uiCallsitesLocalOrDefault:
      /formatTimestamp\(\s*0\s*,\s*"local"\s*\)/.test(ui) ||
      /formatTimestamp\([^,()]*\)/.test(ui),
    // the .map(formatTimestamp) HOF use was NOT rewritten to a wrong-arity
    // call (it stays a bare reference; arity is the compiler's to flag)
    hofCallsiteNotMisedited: /\.map\(\s*formatTimestamp\s*\)/.test(ui)
  };
}

export interface T01Batch {
  modules: { path: string; moduleId: string }[];
}
export interface T01CriteriaInput {
  commitReturnedOk: boolean;
  validateAfterCommitClean: boolean;

  kind: string;
}

  batch: T01Batch,
  srcRoot: string,
  input: T01CriteriaInput
): T01Criteria {
  const rendered = new Map<string, string>();
  for (const m of batch.modules) {
    rendered.set(
      path.relative(srcRoot, m.path).replaceAll("\\", "/"),
      renderModule(db, m.moduleId)
    );

  const text = evaluateT01TextCriteria(rendered);
  const ops = db
    .prepare(`SELECT tx_id, kind FROM operations`)
    .all() as OperationRow[];
  return {
    commitReturnedOk: input.commitReturnedOk === true,
    validateAfterCommitClean: input.validateAfterCommitClean === true,
    ...text,
    operationRowAppended: ops.some(
      (o) => o.tx_id === input.txId && o.kind === "AddParameter"
    )
  };
}

  const loaded = loadModule(db, moduleId);
  return renderWithSourceMap(loaded.module, loaded.children).text;
}
function mustGet(map: Map<string, string>, key: string): string {
  const v = map.get(key);
  if (v === undefined) throw new Error(`Missing rendered module: ${key}`);
  return v;
}
```

```ts
export interface T05Criteria {
  commitReturnedOk: boolean;
  validateAfterCommitClean: boolean;
  comparisonIsHalfOpen: boolean;
  noClosedIntervalRemains: boolean;
  testFileByteIdentical: boolean;
  operationRowAppended: boolean;
}

export type T05TextCriteria = Pick<
  T05Criteria,
  "comparisonIsHalfOpen" | "noClosedIntervalRemains" | "testFileByteIdentical"
>;

/** The T05 test module key, relative to corpus src/ root. Both adapters
 * key the test file this way so the byte-identical check is symmetric. */
export const T05_TEST_KEY = "__test__/dateRange.test.ts";

/**
 * Pure T05 control criteria. `seedTestText` is the byte-exact seed
 * tests/dateRange.test.ts; the check is applied IDENTICALLY for both
 * configs (BS15-D anti-cheat symmetry) — passed in, never file-read here.
 */
export function evaluateT05TextCriteria(
  modules: Map<string, string>,
  seedTestText: string
): T05TextCriteria {
  const dr = mustGet(modules, "lib/dateRange.ts");
  const testText = modules.get(T05_TEST_KEY);
  return {
    comparisonIsHalfOpen:
      /date\s*>=\s*start\s*&&\s*date\s*<\s*end/.test(dr),
    noClosedIntervalRemains: !/date\s*<=\s*end/.test(dr),
    testFileByteIdentical: testText === seedTestText
  };
}

export interface T05Batch {
  modules: { path: string; moduleId: string }[];
}
export interface T05CriteriaInput {
  commitReturnedOk: boolean;
  validateAfterCommitClean: boolean;
  txId: string;
  seedTestText: string;
}
interface OperationRow {
  tx_id: string;
  kind: string;
}

export function evaluateT05Criteria(
  db: Db,
  batch: T05Batch,
  srcRoot: string,
  input: T05CriteriaInput
): T05Criteria {
  const rendered = new Map<string, string>();
  for (const m of batch.modules) {
    rendered.set(
      path.relative(srcRoot, m.path).replaceAll("\\", "/"),
      renderModule(db, m.moduleId)
    );
  }
  // Substrate ingests src/ only; the test file is unchanged by the
  // substrate (it cannot touch files), so feed the seed text under the
  // shared key to keep the byte-identical check symmetric with baseline.
  rendered.set(T05_TEST_KEY, input.seedTestText);
  const text = evaluateT05TextCriteria(rendered, input.seedTestText);
  const ops = db
    .prepare(`SELECT tx_id, kind FROM operations`)
    .all() as OperationRow[];
  return {
    commitReturnedOk: input.commitReturnedOk === true,
    validateAfterCommitClean: input.validateAfterCommitClean === true,
    ...text,
    operationRowAppended: ops.some(
      (o) => o.tx_id === input.txId && o.kind === "ReplaceBody"
    )
  };
}

function renderModule(db: Db, moduleId: string): string {
  const loaded = loadModule(db, moduleId);
  return renderWithSourceMap(loaded.module, loaded.children).text;
}
function mustGet(map: Map<string, string>, key: string): string {
  const v = map.get(key);
  if (v === undefined) throw new Error(`Missing rendered module: ${key}`);
  return v;
}
```

```ts
import type { BenchTaskId } from "./tasks";

export interface SuiteReport {
  model: string;
  n: number;
  generatedAt: string;
  perTask: Record<string, ReturnType<typeof buildReport>>;
  patternHolds: boolean;
  patternNote: string;
  totalCostUsd: number;
}

const STRUCTURAL: BenchTaskId[] = ["T01", "T03", "T08"];
const CONTROL: BenchTaskId = "T05";

export function buildSuiteReport(input: {
  model: string;
  n: number;
  perTask: Record<string, { substrate: TrialMetrics[]; baseline: TrialMetrics[] }>;
  totalCostUsd: number;
}): SuiteReport {
  const perTask: Record<string, ReturnType<typeof buildReport>> = {};
  for (const [id, runs] of Object.entries(input.perTask)) {
    perTask[id] = buildReport({
      task: "T03", // single-task block reused; label overridden in markdown
      model: input.model,
      n: input.n,
      substrate: runs.substrate,
      baseline: runs.baseline,
      totalCostUsd: 0
    });
  }
  const sep = (id: string): boolean =>
    !overlaps(
      perTask[id]!.substrate.totalTokens.values,
      perTask[id]!.baseline.totalTokens.values
    );
  const structuralSeparate = STRUCTURAL.every((id) => perTask[id] && sep(id));
  const controlSeparates = perTask[CONTROL] ? sep(CONTROL) : false;
  const patternHolds = structuralSeparate && !controlSeparates;
  return {
    model: input.model,
    n: input.n,
    generatedAt: new Date().toISOString(),
    perTask,
    patternHolds,
    patternNote:
      "The pattern holds iff the structural tasks (T01,T03,T08) separate AND " +
      "the control (T05) does not; overlapping structural tasks or a " +
      "separated control are reported as stated, not massaged (BS-Bench-D).",
    totalCostUsd: input.totalCostUsd
  };
}

export function renderSuiteMarkdown(report: SuiteReport): string {
  const lines: string[] = [];
  lines.push("# Phase 1.5 four-task benchmark — substrate vs. baseline");
  lines.push("");
  lines.push(`Model: \`${report.model}\` · N per config: ${report.n}`);
  lines.push(`Round total cost: $${report.totalCostUsd.toFixed(2)}`);
  lines.push("");
  for (const [id, r] of Object.entries(report.perTask)) {
    lines.push(`## ${id}`);
    lines.push(renderMarkdown(r).split("\n").slice(2).join("\n"));
    lines.push("");
  }
  lines.push("## cross-task pattern");
  lines.push(report.patternNote);
  lines.push(
    `Observed: pattern ${report.patternHolds ? "HOLDS" : "does NOT hold"} ` +
      "at this N (an observed separation, not a significance claim)."
  );
  return lines.join("\n");
}
```
